SELECT emp_id,CONCAT(last_name,', ',first_name) AS full_name,position.position_shortname, DATE_ADD(appointment_date, INTERVAL 10 YEAR) AS scheduled_ten,DATE_ADD(appointment_date, INTERVAL 15 YEAR) AS scheduled_fifteen,appointment_date
from personnel
INNER JOIN position on personnel.position_id = position.position_id
WHERE ((YEAR(DATE_ADD(appointment_date, INTERVAL 15 YEAR)) = YEAR(CURDATE())) AND
(DATE_ADD(appointment_date, INTERVAL 15 YEAR) < LAST_DAY(DATE_ADD(NOW(), INTERVAL 12-MONTH(NOW()) MONTH))))
ORDER BY appointment_date ASC
